Measures (Dax Formulas)
1) Total Installs = SUM( 'Google_data_cleaned Data set'[installs])
2) Total Revenue = SUM('Google_data_cleaned Data set'[price])
3) Total Apps = DISTINCTCOUNT('Google_data_cleaned Data set'[app])
4) Total Reviews = SUM('Google_data_cleaned Data set'[reviews])
5) Total Installs YoY = 
VAR CurrentYearInstalls = SUM('Google_data_cleaned Data set'[installs])
VAR PreviousYearInstalls = CALCULATE(SUM('Google_data_cleaned Data set'[installs]), SAMEPERIODLASTYEAR('Google_data_cleaned Data set'[Date].[Date]))
RETURN
DIVIDE(CurrentYearInstalls - PreviousYearInstalls, PreviousYearInstalls)
6) Total Revenue YoY = 
VAR CurrentYearRevenue = SUM('Google_data_cleaned Data set'[price])
VAR PreviousYearRevenue = CALCULATE(SUM('Google_data_cleaned Data set'[price]), SAMEPERIODLASTYEAR('Google_data_cleaned Data set'[Date].[Date]))
RETURN
DIVIDE(CurrentYearRevenue - PreviousYearRevenue, PreviousYearRevenue)
7) Total Apps YoY = 
VAR CurrentYearApps = DISTINCTCOUNT('Google_data_cleaned Data set'[app])
VAR PreviousYearApps = CALCULATE(DISTINCTCOUNT('Google_data_cleaned Data set'[app]), SAMEPERIODLASTYEAR('Google_data_cleaned Data set'[Date].[Date]))
RETURN
DIVIDE(CurrentYearApps - PreviousYearApps, PreviousYearApps)
8) Avg Rating YoY = 
VAR CurrentYearAvgRating = AVERAGE('Google_data_cleaned Data set'[rating])
VAR PreviousYearAvgRating = CALCULATE(AVERAGE('Google_data_cleaned Data set'[rating]), SAMEPERIODLASTYEAR('Google_data_cleaned Data set'[Date].[Date]))
RETURN
DIVIDE(CurrentYearAvgRating - PreviousYearAvgRating, PreviousYearAvgRating)
